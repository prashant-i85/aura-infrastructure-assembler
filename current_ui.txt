  <script type="text/babel">
    const { useState, useEffect, useRef } = React;

    const API_BASE = 'http://localhost:8080/api';
    const INSTANCE_TYPES = ['t2.nano', 't2.micro', 't2.small', 't2.medium', 't2.large', 't3.nano', 't3.micro', 't3.small', 't3.medium', 't3.large'];
    const GCP_MACHINE_TYPES = ['e2-micro', 'e2-small', 'e2-medium', 'e2-standard-2', 'e2-standard-4'];
    
    const AWS_REGIONS = [
      { value: 'us-east-1', label: 'US East (N. Virginia)' },
      { value: 'us-east-2', label: 'US East (Ohio)' },
      { value: 'us-west-1', label: 'US West (N. California)' },
      { value: 'us-west-2', label: 'US West (Oregon)' }
    ];
    const GCP_REGIONS = [
      { value: 'us-central1', label: 'US Central (Iowa)' },
      { value: 'us-east1', label: 'US East (S. Carolina)' }
    ];
    
    const REGION_AMIS = {
      'us-east-1': 'ami-0c02fb55956c7d316',
      'us-east-2': 'ami-097a2df4ac947655f',
      'us-west-1': 'ami-0d9858aa3c6322f73',
      'us-west-2': 'ami-0892d3c7ee96c0bf7',
    };

    const PROVISION_STEPS = [
      { id: 'config', label: 'Generating Terraform config' },
      { id: 'init', label: 'Initializing Terraform' },
      { id: 'plan', label: 'Running plan' },
      { id: 'apply', label: 'Provisioning resources' },
      { id: 'output', label: 'Extracting metadata' },
      { id: 'done', label: 'Instance running' },
    ];

    function getStepIndex(message) {
      if (!message) return -1;
      if (message.includes('Generating')) return 0;
      if (message.includes('Initializing')) return 1;
      if (message.includes('plan')) return 2;
      if (message.includes('Applying') || message.includes('provisioning')) return 3;
      if (message.includes('Extracting') || message.includes('metadata')) return 4;
      if (message.includes('successfully')) return 5;
      return -1;
    }

    async function apiFetch(path, options = {}) {
      const res = await fetch(`${API_BASE}${path}`, {
        headers: { 'Content-Type': 'application/json' },
        ...options,
      });
      const data = await res.json();
      if (!res.ok) throw new Error(data.message || data.error || 'Request failed');
      return data;
    }

    function StatusBadge({ status }) {
      const s = (status || '').toLowerCase();
      const cls = {
        pending: 'badge-pending', in_progress: 'badge-in_progress',
        success: 'badge-success', failed: 'badge-failed',
        destroying: 'badge-destroying', destroyed: 'badge-destroyed',
      }[s] || 'badge-pending';
      return <span className={`status-badge ${cls}`}>{status}</span>;
    }

    function StatusTracker({ requestId }) {
      const [data, setData] = useState(null);
      const intervalRef = useRef(null);

      async function fetchStatus(id) {
        try {
          const res = await apiFetch(`/infrastructure/status/${id}`);
          setData(res);
          if (res.status === 'SUCCESS' || res.status === 'FAILED' || res.status === 'DESTROYED') {
            clearInterval(intervalRef.current);
          }
        } catch (e) {
          clearInterval(intervalRef.current);
        }
      }

      useEffect(() => {
        if (!requestId) return;
        fetchStatus(requestId);
        intervalRef.current = setInterval(() => fetchStatus(requestId), 5000);
        return () => clearInterval(intervalRef.current);
      }, [requestId]);

      if (!data) return <div style={{color: 'var(--text-secondary)'}}>Loading status...</div>;
      
      const stepIndex = getStepIndex(data.message);
      
      return (
        <div>
          <div style={{display:'flex',justifyContent:'space-between',marginBottom:'12px'}}>
             <StatusBadge status={data.status} />
             <span className="mono text-muted">{data.requestId.substring(0,8)}...</span>
          </div>
          <div style={{fontSize:'0.85rem', color:'var(--text-secondary)'}}>{data.message}</div>
          
          {(data.status === 'IN_PROGRESS' || data.status === 'PENDING') && (
            <div className="progress-steps">
              {PROVISION_STEPS.map((step, i) => {
                const state = i < stepIndex ? 'done' : i === stepIndex ? 'active' : 'pending';
                return (
                  <div key={step.id} className={`progress-step ${state}`}>
                    <span>{state === 'done' ? '✓' : state === 'active' ? '⟳' : '○'}</span> {step.label}
                  </div>
                );
              })}
            </div>
          )}

          {data.status === 'SUCCESS' && (
            <div className="status-meta">
              <div><label style={{fontSize:'0.75rem'}}>Instance ID</label><div className="mono">{data.instanceId}</div></div>
              <div><label style={{fontSize:'0.75rem'}}>Public IP</label><div className="mono">{data.publicIp}</div></div>
            </div>
          )}
        </div>
      );
    }

    function ProvisionDashboard() {
      const [form, setForm] = useState({
        provider: 'AWS', resourceType: 'COMPUTE', instanceType: 't2.micro',
        region: 'us-east-1', amiId: REGION_AMIS['us-east-1'], instanceName: '', keyPairName: ''
      });
      const [pricing, setPricing] = useState(null);
      const [loading, setLoading] = useState(false);
      const [trackingId, setTrackingId] = useState(null);
      const [manualId, setManualId] = useState('');

      useEffect(() => {
          fetchPricing(form.resourceType);
      }, [form.resourceType]);

      async function fetchPricing(type) {
          try {
              const res = await fetch(`${API_BASE}/pricing/compare?resourceType=${type}`);
              setPricing(await res.json());
          } catch(e) {}
      }

      function update(f, v) {
        if (f === 'region' && form.provider === 'AWS' && form.resourceType === 'COMPUTE') {
          setForm(p => ({ ...p, region: v, amiId: REGION_AMIS[v] || '' }));
        } else if (f === 'provider') {
          let r = v === 'AWS' ? 'us-east-1' : 'us-central1';
          let t = v === 'AWS' ? 't2.micro' : 'e2-micro';
          setForm(p => ({ ...p, provider: v, region: r, instanceType: t, amiId: v === 'AWS' ? REGION_AMIS[r] : 'debian-cloud/debian-11' }));
        } else {
          setForm(p => ({ ...p, [f]: v }));
        }
      }

      async function submit(e) {
        e.preventDefault();
        setLoading(true);
        try {
          const res = await apiFetch('/infrastructure/provision', {
            method: 'POST', body: JSON.stringify(form),
          });
          setTrackingId(res.requestId);
        } catch (err) {
          alert('Error: ' + err.message);
        } finally {
          setLoading(false);
        }
      }

      return (
        <form onSubmit={submit} className="dashboard-grid">
          {/* Left Column */}
          <div className="col-left">
            <div className="card">
              <div className="card-title">
                 <span className="icon">⚙️</span> Setup & Config
              </div>
              <div className="text-secondary" style={{fontSize: '0.9rem', marginBottom: '1rem'}}>
                 Configure baseline settings for the deployment.
              </div>
            </div>

            <div className="card">
              <div className="card-title"><span className="icon">🚀</span> Provision Request</div>
              
              <div className="form-group">
                <label>Cloud Provider</label>
                <select className="form-control" value={form.provider} onChange={e => update('provider', e.target.value)}>
                  <option value="AWS">AWS</option><option value="GCP">GCP</option>
                </select>
              </div>
              
              <div className="form-group">
                <label>Resource Type</label>
                <select className="form-control" value={form.resourceType} onChange={e => update('resourceType', e.target.value)}>
                  <option value="COMPUTE">Compute</option><option value="STORAGE">Storage</option>
                </select>
              </div>

              {form.resourceType === 'COMPUTE' && (
                <div className="form-group">
                  <label>Instance Type</label>
                  <select className="form-control" value={form.instanceType} onChange={e => update('instanceType', e.target.value)}>
                    {form.provider === 'AWS' ? INSTANCE_TYPES.map(t=><option key={t}>{t}</option>) : GCP_MACHINE_TYPES.map(t=><option key={t}>{t}</option>)}
                  </select>
                </div>
              )}

              <div className="form-group">
                <label>Region</label>
                <select className="form-control" value={form.region} onChange={e => update('region', e.target.value)}>
                  {form.provider === 'AWS' ? AWS_REGIONS.map(r=><option key={r.value} value={r.value}>{r.label}</option>) : GCP_REGIONS.map(r=><option key={r.value} value={r.value}>{r.label}</option>)}
                </select>
              </div>

              <div style={{marginTop: '1.5rem'}}>
                <button type="submit" className="btn btn-primary" disabled={loading}>
                  {loading ? <><div className="spinner" /> Submitting</> : 'Provision Instance'}
                </button>
              </div>
            </div>
          </div>

          {/* Mid Column */}
          <div className="col-mid">
            <div className="card">
               <div className="card-title"><span className="icon">📦</span> Image & Identity</div>
               
               <div className="ami-logo-block">
                 <div className="ami-icon-box">{form.provider === 'AWS' ? '🐧' : '🪶'}</div>
                 <div>
                   <div style={{fontWeight:600}}>{form.provider === 'AWS' ? 'Ubuntu 22.04 LTS' : 'Debian 11'}</div>
                   <div className="text-secondary" style={{fontSize:'0.8rem'}}>{form.provider === 'AWS' ? 'Provided by AWS Marketplace' : 'Provided by Google Cloud'}</div>
                 </div>
               </div>

               {form.resourceType === 'COMPUTE' && (
                  <div className="form-group">
                    <label>Image / AMI ID</label>
                    <input className="form-control" value={form.amiId} onChange={e => update('amiId', e.target.value)} />
                  </div>
               )}

               <div className="form-group">
                 <label>Instance Name</label>
                 <input className="form-control" required placeholder="aura-server-01" value={form.instanceName} onChange={e => update('instanceName', e.target.value)} />
               </div>

               <div className="form-group">
                 <label>Key Pair Name (optional)</label>
                 <input className="form-control" placeholder="my-keypair" value={form.keyPairName} onChange={e => update('keyPairName', e.target.value)} />
               </div>
            </div>
          </div>

          {/* Right Column */}
          <div className="col-right">
             <div className="card">
               <div className="card-title"><span className="icon">💰</span> Pricing Compare</div>
               {pricing ? (
                 <div>
                   <div className="pricing-row">
                     <span className="text-secondary">AWS ({pricing.awsDetails})</span>
                     <span style={{fontWeight:600}}>${pricing.awsPricePerHour.toFixed(3)}/hr</span>
                   </div>
                   <div className="pricing-row">
                     <span className="text-secondary">GCP ({pricing.gcpDetails})</span>
                     <span style={{fontWeight:600}}>${pricing.gcpPricePerHour.toFixed(3)}/hr</span>
                   </div>
                   <div className="pricing-row" style={{marginTop:'8px', paddingTop:'12px'}}>
                     <span className="text-primary" style={{fontWeight:500}}>Recommendation</span>
                     <span style={{color:'var(--accent-green)', fontWeight:600}}>{pricing.recommendation}</span>
                   </div>
                 </div>
               ) : <div className="text-secondary" style={{fontSize:'0.9rem'}}>Loading pricing...</div>}
             </div>

             <div className="card">
               <div className="card-title"><span className="icon">📡</span> Live Status</div>
               {trackingId ? <StatusTracker requestId={trackingId} /> : <div className="text-secondary" style={{fontSize:'0.9rem'}}>Submit a request to see live tracking.</div>}
             </div>
          </div>

          {/* Bottom Row */}
          <div className="bottom-row">
             <div className="bottom-track-card">
                <div style={{fontSize:'1.2rem'}}>🔍</div>
                <div style={{flex: 1}}>
                  <div style={{fontWeight:600}}>Track existing request</div>
                  <div className="text-secondary" style={{fontSize:'0.85rem'}}>Paste a Request ID to monitor live status</div>
                </div>
                <input className="form-control" style={{width: '300px'}} placeholder="Request ID..." value={manualId} onChange={e=>setManualId(e.target.value)} />
                <button type="button" className="btn btn-ghost" style={{width: 'auto'}} onClick={() => setTrackingId(manualId)} disabled={!manualId}>Track</button>
             </div>
          </div>

        </form>
      );
    }

    function App() {
      const [activeTab, setActiveTab] = useState('Dashboard');
      const [theme, setTheme] = useState('light');
      const [healthStatus, setHealthStatus] = useState('check');
      
      useEffect(() => {
        document.documentElement.setAttribute('data-theme', theme);
      }, [theme]);
      
      useEffect(() => {
        apiFetch('/health')
          .then(h => setHealthStatus(h.status === 'UP' ? 'up' : 'down'))
          .catch(() => setHealthStatus('down'));
      }, []);

      const tabs = ['Dashboard', 'People', 'Hiring', 'Devices'];

      return (
        <>
          <div className="topbar-wrapper">
             <header className="topbar">
               <div style={{fontWeight: 700, fontSize:'1.1rem', marginRight: '2rem'}}>AURA</div>
               
               <nav className="topbar-nav">
                 {tabs.map(t => (
                   <div key={t} className={`nav-link ${activeTab === t ? 'active' : ''}`} onClick={() => setActiveTab(t)}>
                     {t}
                   </div>
                 ))}
               </nav>

               <div className="topbar-right">
                 <div className={`health-badge ${healthStatus}`}>
                   <span className="health-dot"/> {healthStatus === 'up' ? 'Online' : 'Offline'}
                 </div>
                 <button className="icon-btn" onClick={() => setTheme(t => t === 'light' ? 'dark' : 'light')}>
                   {theme === 'light' ? '🌙' : '☀️'}
                 </button>
                 <button className="icon-btn">⚙️</button>
                 <button className="icon-btn">👤</button>
               </div>
             </header>
          </div>

          <main className="main">
             {activeTab === 'Dashboard' ? <ProvisionDashboard /> : (
                <div style={{textAlign:'center', marginTop:'4rem', color:'var(--text-secondary)'}}>
                  <h2>{activeTab} Module</h2>
                  <p>In development.</p>
                </div>
             )}
          </main>
        </>
      );
    }

    ReactDOM.createRoot(document.getElementById('root')).render(<App />);
  </script>
